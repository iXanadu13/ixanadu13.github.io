#include <bits/stdc++.h>
using namespace std;
const int N = 8;
int n,m,t,sx,sy,ex,ey;
char mp[N][N];
bool tag;
int nxt[4][2]={1,0,0,1,-1,0,0,-1};
void dfs(int x,int y,int step){
    if(tag || step>t) return;
    if(x==ex&&y==ey&&step==t){
        tag= true;
        return;
    }
    for(int i=0;i<4;i++){
        int nx=x+nxt[i][0];
        int ny=y+nxt[i][1];
        if(nx>=1&&nx<=n&&ny>=1&&ny<=m && mp[nx][ny]!='X'){
            mp[nx][ny]='X';
            dfs(nx,ny,step+1);
            mp[nx][ny]='.';
        }
    }
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    while (cin>>n>>m>>t,n||m||t){
        int wall=0;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                cin>>mp[i][j];
                if(mp[i][j]=='X') ++wall;
                else if(mp[i][j]=='S') sx=i,sy=j;
                else if(mp[i][j]=='D') ex=i,ey=j;
            }
        }
        if(n*m-wall<=t){
            cout<<"NO\n";
            continue;
        }
        if((abs(sx+sy-ex-ey)&1)!=(t&1)){
            cout<<"NO\n";
            continue;
        }
        if (t<abs(sx-ex)+abs(sy-ey)-2) {
            cout<<"NO\n";
            continue;
        }
        tag=false;
        mp[sx][sy]='X';
        dfs(sx,sy,0);
        cout<<(tag?"YES":"NO")<<'\n';
    }
}
