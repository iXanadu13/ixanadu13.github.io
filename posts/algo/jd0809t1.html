using namespace std;
using i64 = long long;

void solve() {
    int n, k;
    cin >> n >> k;
    map<int, int> cnt;
    set<int> vis;
    vector<int> arr(n);
    for (auto &e : arr){
        cin >> e;
        ++cnt[e];
    }

    vector<int> ans;
    for (auto x : arr) {
        if (!vis.count(x)) {
            while (!ans.empty() && ans.back() > x && cnt[ans.back()]) {
                vis.erase(ans.back());
                ans.pop_back();
            }
            vis.insert(x);
            ans.push_back(x);
        }
        --cnt[x];
    }

    for (size_t i = 0; i < ans.size(); ++i) {
        cout << ans[i];
        if (i != ans.size() - 1) cout << ' ';
    }
}
